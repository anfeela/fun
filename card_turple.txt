shengcheng 0-4 de suijizhengshu,ranhou duqu duiying suoyin            n = random.randint(0,4)            ##print('choudao:{ }'.format(card_turple[n]))            print(card_turple[n])            print('你抽到了: {}'.format(card_turple[n]))      ##
